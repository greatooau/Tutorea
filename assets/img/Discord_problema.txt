#include <windows.h>
#include <iostream>
#include <fstream>
#include <stdlib.h>
#include <string>
#include "resource.h"

using namespace std;

LRESULT CALLBACK callbackpantalladeinicio(HWND ventana, UINT mensaje, WPARAM wparam, LPARAM lparam)
{
    switch (mensaje)
    {
    case WM_INITDIALOG:

        break;
    case WM_COMMAND:

       
        break;

    case WM_CLOSE:
        MessageBox(ventana, L"Buena Suerte", L"Despedida", MB_OK | MB_ICONASTERISK);
        PostQuitMessage(0);
        break;
    default:
        break;
    }

    return DefWindowProc(ventana, mensaje, wparam, lparam);
}

HWND ghDlg = 0;

int WINAPI WinMain(HINSTANCE a, HINSTANCE b, LPSTR c, int d)
{
    ghDlg = CreateDialog(a, MAKEINTRESOURCE(VENTANA_EJECUCION), 0, callbackpantalladeinicio);
    ShowWindow(ghDlg, d);
    MSG msg;
    ZeroMemory(&msg, sizeof(MSG));
    while (GetMessage(&msg, 0, 0, 0)) {
        if (ghDlg == 0 || !IsDialogMessage(ghDlg, &msg)) {
            TranslateMessage(&msg);
            DispatchMessage(&msg);
        }
    }

    return (int)msg.wParam;

}
